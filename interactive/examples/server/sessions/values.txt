# Structured (ADT) values + time-scheduled input.
install echo examples/server/programs/echo.ddp

# A Variant payload at the current epoch (0): a tagged tuple, the shape an AST node takes.
# (values are one token — write terms without spaces)
feed echo 0 inject(2,tuple(1,2))
# A List value scheduled for a FUTURE logical time (3) — appears only once the
# input frontier passes 3 (i.e., after the 4th tick).
feed echo 0 list(7,8,9) time=3

tick
tick
tick
tick

# A malformed value: the server reports it and keeps running (robustness).
feed echo 0 inject(oops)
list
exit
